allianceaction = {
	condition = { ### Союзы только внутри империи (для СРИ)
		tooltip = not_alliance_with_non_HRE_1
		potential = {
			is_part_of_hre = yes
			is_emperor = no
			NOT = { tag = PRU }
		}
		allow = {
			FROM = {
				OR = {
					is_part_of_hre = yes
				}
			}
		}
	}
	condition = { ### Союзы только внутри империи (для не СРИ)
		tooltip = not_alliance_with_non_HRE_2
		potential = {
			is_part_of_hre = no
		}
		allow = {
			FROM = {
				OR = {
					is_part_of_hre = no
					is_emperor = yes
					tag = PRU
				}
			}
		}
	}
	
	condition = { ### Княжества Руси, которые являются данниками, не могут заключать союзы

		potential = {
			culture_group = east_slavic
			NOT = { tag = NOV }
			NOT = { government_rank = 3 }
		}
		allow = {
			FROM = {
				OR = {
					culture_group = east_slavic
				}
			}
		}
	}
	
	condition = { ### Нельзя заключать союзы с князьями Руси
		tooltip = not_alliance_with_non_russian_princes_2
		potential = {
			NOT = { culture_group = east_slavic }
		}
		allow = {
			FROM = {
				OR = {
					NOT = { culture_group = east_slavic }
					tag = NOV
					AND = {

				}
			}
		}
