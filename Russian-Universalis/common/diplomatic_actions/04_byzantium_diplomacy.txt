warningaction = {
	
	condition = { ### Ограничение возможности угрожать другим гос-вам за Византию (BYZ)
		tooltip = BYZ_CANNOT_WARNINGACTION
		potential = {
			tag = BYZ 
			has_country_modifier = byz_mission_army_crisis_one
		}
		allow = {
			always = no 
		}
	}
}

threaten_war  = {
	
	condition = { ### Ограничение возможности угрозой войны, отобрать территорию, у других гос-в за Византию (BYZ)
		tooltip = BYZ_CANNOT_REQUEST_THREATEN_WAR
		potential = {
			tag = BYZ 
			OR = {
				has_country_modifier = byz_mission_army_crisis_one
				has_country_modifier = byz_mission_army_crisis_two
				has_country_modifier = byz_mission_army_crisis_three
			}
		}
		allow = {
			always = no 
		}
	}
}

guaranteeaction = {
	
	condition = { ### Ограничение возможности гарантировать независимость другим гос-вам за Византию (BYZ)
		tooltip = BYZ_CANNOT_GUARANTEEACTION
		potential = {
			tag = BYZ 
			has_country_modifier = byz_mission_army_crisis_one
		}
		allow = {
			always = no 
		}
	}
}

royal_marriage = {
	
	condition = { ### Стандартное условие
		tooltip = MARWAR
		potential = {
		}
		allow = {
			NOT = { war_with = FROM }
		}
	}
	
	condition = { ### Ограничение возможности заключать династические браки за Византию (BYZ)
		tooltip = BYZ_CANNOT_REQUEST_ROYAL_MARRIAGE
		potential = {
			tag = BYZ
			OR = {
				AND = {
					has_country_modifier = byz_mission_diplomatic_crisis_one
					NOT = { has_country_modifier = byz_mission_diplomatic_crisis_two }
				}
				AND = {
					has_country_modifier = byz_mission_diplomatic_crisis_one
					NOT = { has_country_modifier = stt_byz_bride }
				}
			}
		}
		allow = {
			always = no 
		}
	}
	
}

vassalaction = {
	
	condition = { ### Стандартное условие
		tooltip = VASSALINVALID
		potential = {
		}
		allow = {
			alliance_with = FROM
			FROM = {
				has_opinion = {
					who = ROOT
					value = 190
				}
			}
			FROM = { is_at_war = no }
		}
	}
	
	condition = { ### Ограничение возможности мирной вассализации других держав за Византию (BYZ)
		tooltip = BYZ_CANNOT_REQUEST_VASSAL_ACTION
		potential = {
			tag = BYZ
			OR = {
				has_country_modifier = byz_mission_diplomatic_crisis_one
				has_country_modifier = byz_mission_diplomatic_crisis_two
			}
		}
		allow = {
			always = no 
		}
	}
	
}
