namespace = vic_institutions

#Imperialism
country_event = {
	id = vic_institutions.1
	title = vic_institutions.1.t
	desc = vic_institutions.1.d
	picture = imperialism_event_picture

	is_triggered_only = yes
	

	trigger = {
		always = yes			#
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = vic_institutions.1.a
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_prestige = 20
		FROM = {
			add_permanent_province_modifier = {
				name = "birthplace_of_imperialism"
				duration = -1
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					knows_country = ROOT
				}
				country_event = { id = vic_institutions.100 }
			}
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					NOT = { knows_country = ROOT }
				}
				country_event = { id = vic_institutions.101 }
			}
		}
	}
}
