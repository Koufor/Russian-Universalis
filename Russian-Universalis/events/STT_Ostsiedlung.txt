# Изменение культуры
country_event = {
	id = stt_ostsiedlung.2
	title = stt_ostsiedlung.2.t
	desc = stt_ostsiedlung.2.d
	picture = CITY_VIEW_eventPicture
	
	#is_triggered_only = yes 
	#hidden = yes
	
	trigger = {
		any_owned_province = {
			check_variable = { which = conversion_var value = 100 }
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
		
	}
	option = {
		name = stt_ostsiedlung.2.1
		every_owned_province = {
			limit = { check_variable = { which = conversion_var value = 100 } }
			# Смена культуры в зависимости от области
			if = {
				limit = { 
					owner = { culture_group = germanic }
					OR = {
						area = vorpommern_area
						area = hinter_pommern_area
					}
				}
				change_culture = pommeranian
			}
			if = {
				limit = { 
					owner = { culture_group = germanic }
					OR = {
						area = mittelmark_area
						area = neumark_area
						area = south_saxony_area
						area = erzgebirge_area
					}
				}
				change_culture = saxon
			}
			if = {
				limit = { 
					owner = { culture_group = germanic }
					OR = {
						area = livonia_area
						area = east_livonia_area
						area = estonia_ingria_area
						area = east_estonia_area 
						area = curonia_area
						area = ingria_area
					}
				}
				change_culture = ostzian
			}
			if = {
				limit = { 
					owner = { culture_group = germanic }
					OR = {
						area = west_prussia_area
						area = east_prussia_area
						area = covno_area
					}
				}
				change_culture = prussian
			}
			if = {
				limit = { 
					OR = {
						area = silesia_area
						area = upper_silesia_area
						province_id = 3732 # Освенцим
						province_id = 3244 # Цешин
					}
				}
				change_culture = german_silesian
			}
			if = {
				limit = { 
					owner = { culture_group = germanic }
					OR = { 
						area = wielkopolska_area
						area = lithuania_area
						area = ingria_area
						area = pskov_area 
						area = kalish_area
						area = mazovia_area
						area = novgorod_area
					}
				}
				change_culture = owner
			}
			
			# Чистка флагов, модификаторов и переменных
			hidden_effect = { 
				clr_province_flag = province_of_ostsiedlung
				remove_province_modifier = promote_ostsiedlung_1
				set_variable = { which = conversion_var value = 0 }
				add_province_modifier = {
					name = promote_ostsiedlung_2
					duration = 3650
				}
			}
			
			# Вопрос смены религии (рандом)
			if = {
				limit = { NOT = { religion = owner } }
				random_list = {
					90 = { add_unrest = 10 }
					5 = { change_religion = owner add_unrest = 5 }
					4 = { }
					1 = { change_religion = owner }
				}
			}
			
			# Соседние провинции получат флаг для продолжения распространения миграции
			every_neighbor_province = {
				limit = { 
					OR = {
						owner = { culture = silesian }
						owner = { culture = czech }
						owner = { culture_group = germanic }
					}
					NOT = { has_province_modifier = promote_ostsiedlung_1 }
					NOT = { has_province_modifier = promote_ostsiedlung_2 }
					province_group = german_ostsiedlung_group
				}
				set_province_flag = can_to_ostsiedlung 
			}
		}
		#hidden_effect = { clr_country_flag = promote_ostsiedlung }
	}	
}

# Вспомогательный ивент (убирает ограничение на ассимиляцию культур)
country_event = {
	id = stt_ostsiedlung.4
	title = stt_ostsiedlung.4.t
	desc = stt_ostsiedlung.4.d
	picture = CITY_VIEW_eventPicture
	
	#is_triggered_only = yes 
	hidden = yes
	
	trigger = {
		OR = {
			any_country = {
				has_country_flag = country_of_ostsiedlung
				NOT = { culture_group = germanic }
			}
			AND = {
				any_country = { has_country_flag = country_of_ostsiedlung }
				is_year = 1800
			}
			AND = {
				any_country = { has_country_flag = country_of_ostsiedlung }
				all_owned_province = { culture_group = germanic }
			}
		}
	}
	
	option = {
		name = stt_ostsiedlung.4.1
		
		hidden_effect = { clr_country_flag = country_of_ostsiedlung }
	}	
}

# Выдача флага Country_of_Ostsiedlung
country_event = {
	id = stt_ostsiedlung.7
	title = stt_ostsiedlung.7.t
	desc = stt_ostsiedlung.7.d
	picture = plato_book_event_picture
	
	hidden = yes

	trigger = {
		any_country = {
			culture_group = germanic
			any_owned_province = { 
				NOT = { culture_group = owner } 
				province_group = german_ostsiedlung_group
			}
		}
	}
	option = {
		name = stt_ostsiedlung.7.1
		every_country = {
			limit = {
				culture_group = germanic
				any_owned_province = { 
					NOT = { culture_group = owner } 
					province_group = german_ostsiedlung_group
				}
			}
			set_country_flag = country_of_ostsiedlung
			add_country_modifier = {
				name = is_ostsiedlung
				duration = -1
			}
		}
	}
}
