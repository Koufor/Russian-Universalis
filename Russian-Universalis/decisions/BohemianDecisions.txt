country_decisions = {

	include_silesia_into_hre = { # Включить Силезию в состав СРИ
		potential = {
			tag = BOH
			NOT = { is_year = 1330 }
			any_province = {
				area = upper_silesia_area 
				is_part_of_hre = no			
			}
			any_province = {
				area = silesia_area 
				is_part_of_hre = no			
			}
		}
		allow = {
			is_emperor = yes
			OR = {
				AND = {
					custom_trigger_tooltip = {
						tooltip = can_includ_silesia_to_hre
						has_global_flag = silesia_can_including # Флаг можно получить входе Польско-Тевтонского конфликта
					}
					OR = {
						diplomatic_reputation = 2
						AND = {
							advisor = statesman
							dip_power = 150
						}
					}
					imperial_influence = 10
				}
				AND = {
					upper_silesia_area = {
						type = all
						country_or_non_sovereign_subject_holds = ROOT
					}
					silesia_area = {
						type = all
						country_or_non_sovereign_subject_holds = ROOT
					}
				}
			}
		}
		effect = {
			add_prestige = 5
			custom_tooltip = including_silesia_to_hre
			hidden_effect = {
				every_province = {
					limit = { area = upper_silesia_area is_part_of_hre = no }
					set_in_empire = yes
				}
				every_province = {
					limit = { area = silesia_area is_part_of_hre = no }
					set_in_empire = yes
				}
				if = {
					limit = { 
						OR = {
							3732 = { is_part_of_hre = no }
							3244 = { is_part_of_hre = no }
						}
					}
					3732 = { set_in_empire = yes }
					3244 = { set_in_empire = yes }
				}
			}
			if = {
				limit = { has_global_flag = silesia_can_including }
				clr_global_flag = silesia_can_including
				add_dip_power = -150
				
			}
			add_imperial_influence = 10
		}
		ai_will_do = {
			factor = 1
		}
	}
}
